#include <xtd/xtd>
using namespace xtd;
using namespace xtd::drawing;
using namespace xtd::forms;
class form_main : public form {
public:
static auto main() {
application::run(form_main());
}
form_main() = default;
protected:
void on_paint(paint_event_args& e) {
form::on_paint(e);
auto g = e.graphics();
g.clear(colors::white());
draw_shadowed_box(g, 10, 10, 200, 100);
}
private:
void draw_shadowed_box(graphics& g, int x, int y, int width, int height) {
auto brush = solid_brush(colors::black());
auto region = drawing::region();
region.exclude(rectangle {x, y, width, height});
region.intersect(rectangle {x + 11, y + 11, width, height});
g.fill_region(brush, region);
//g.fill_rectangle(solid_brush(colors::green()), x, y, width, height);
g.draw_rectangle(pen(colors::red(), 2), x, y, width, height);
}
};
startup_(form_main);
