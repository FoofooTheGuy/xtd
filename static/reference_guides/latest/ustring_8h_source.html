719 template<typename ...args_t>
720 static ustring format(const ustring& fmt, args_t&& ... args) {
721 ustring result;
722 size_t index = 0;
723 std::vector<__format_information<char>> formats;
724 auto begin_format_iterator = fmt.end();
725 auto end_format_iterator = fmt.end();
726 for (auto iterator = fmt.begin(); iterator != fmt.end(); ++iterator) {
727 if (*iterator == '{') {
728 if (++iterator == fmt.end())
729 __throw_ustring_format_exception_open_bracket();
730 if (*iterator == '{')
731 result += *iterator;
732 else {
733 begin_format_iterator = iterator;
734 while (iterator != fmt.end() && *iterator != '}') ++iterator;
735 if (iterator == fmt.end())
736 __throw_ustring_format_exception_open_bracket();
737 end_format_iterator = iterator;
738 __format_information<char> fi;
739 fi.location = result.size();
740 std::string format_str {begin_format_iterator, end_format_iterator};
741 if (format_str.size() == 0)
742 fi.index = index++;
743 else {
744 size_t index_alignment_separator = ustring(format_str).index_of(',');
745 size_t index_format_separator = ustring(format_str).index_of(u8':');
746
747 if (index_alignment_separator != std::string::npos && index_format_separator != std::string::npos && index_alignment_separator > index_format_separator)
748 index_alignment_separator = std::string::npos;
749
750 if (index_alignment_separator != std::string::npos)
751 fi.alignment = format_str.substr(index_alignment_separator + 1, index_format_separator != std::string::npos ? index_format_separator - index_alignment_separator - 1 : std::string::npos);
752
753 if (index_format_separator != std::string::npos)
754 fi.format = format_str.substr(index_format_separator + 1);
755
756 if (index_alignment_separator == 0 || index_format_separator == 0)
757 fi.index = index++;
758 else {
759 std::string index_str;
760 if (index_alignment_separator != std::string::npos)
761 index_str = format_str.substr(0, index_alignment_separator);
762 else if (index_format_separator != std::string::npos)
763 index_str = format_str.substr(0, index_format_separator);
764 else
765 index_str = std::move(format_str);
766 try {
767 for (auto c : index_str)
768 if (!std::isdigit(c)) __throw_ustring_format_exception_start_colon();
769 fi.index = std::stoi(index_str);
770 } catch (...) {
771 __throw_ustring_format_exception_start_colon();
772 }
773 }
774 }
775 formats.push_back(fi);
776 }
777 } else if (*iterator == '}') {
778 if (++iterator == fmt.end()) {
779 __throw_ustring_format_exception_close_bracket();
780 break;
781 }
782 if (*iterator != '}') {
783 __throw_ustring_format_exception_close_bracket();
784 break;
785 }
786 result += *iterator;
787 } else
788 result += *iterator;
789 }
790
791 __ustring_extract_format_arg(result, formats, std::forward<args_t>(args)...);
792 return result.c_str();
793 }

1665void __ustring_extract_format_arg(std::string& fmt, size_t& index, std::vector<__format_information<char>>& formats, arg_t&& arg) {
1666 size_t offset = 0;
1667 for (auto& format : formats) {
1668 format.location += offset;
1669 if (format.index == index) {
1670 xtd::ustring arg_str = format.format.empty() ? __format_stringer<char, arg_t>(arg) : xtd::to_string(arg, format.format);
1671
1672 if (!format.alignment.empty()) {
1673 xtd::int32 alignment = 0;
1674 try {
1675 alignment = std::stoi(format.alignment);
1676 } catch (...) {
1677 __throw_ustring_format_exception();
1678 }
1679 if (alignment > 0) arg_str = arg_str.pad_left(alignment);
1680 else if (alignment < 0) arg_str = arg_str.pad_right(-alignment);
1681 }
1682 fmt.insert(format.location, arg_str);
1683 offset += arg_str.size();
1684 }
1685 }
1686 ++index;
