195 static enum_type parse_flags(const xtd::ustring& value, bool ignore_case) {
196 std::vector<xtd::ustring> values = value.split({','});
197 for (xtd::ustring& str : values)
198 str = str.trim_start(' ').trim_end(' ');
199
200 if (values.size() == 1) {
201 for (auto item : enum_object<enum_type>().entries()) {
202 if (xtd::ustring::compare(value, item.second, ignore_case) == 0)
203 return to_enum(item.first);

211 for (auto item : enum_object<enum_type>().entries()) {

213 found = true;

236 xtd::ustring to_string_flags() const noexcept {
237 auto iterator = std::find_if(entries().begin(), entries().end(), [&](auto value)->bool {return value.first == value_;});
238 if (to_int(value_) == 0 && iterator == entries().end()) return "0";
239
240 iterator = std::find_if(entries().begin(), entries().end(), [&](auto value)->bool {return value.first == to_enum(0);});
241 if (to_int(value_) == 0) return iterator == entries().end() ? "0" : iterator->second;
242
243 xtd::ustring str;
244 int64 rest = to_int(value_);
245 enum_collection<enum_type> reversed = entries();
246 std::reverse(reversed.begin(), reversed.end());
247
248 for (auto item : reversed) {
249 if (to_int(item.first) != 0 && (rest & to_int(item.first)) == to_int(item.first)) {
250 rest -= to_int(item.first);
251 if (!xtd::ustring::is_empty(str)) str = ", " + str;
252 str = item.second + str;
253 }
254 }
255
256 if (str.empty() || rest > 0) return xtd::ustring::format("{}", to_int(value_));
257
258 return str;
259 }
